SELECT observation_index FROM capstone.synthetic_sensor_messages_send_queue
EXCEPT
SELECT observation_index FROM capstone.synthetic_sensor_messages_consumed_stage;


WITH send_counts AS (
    SELECT
        observation_index,
        COUNT(*) AS send_row_count
    FROM capstone.synthetic_sensor_messages_send_queue
    GROUP BY observation_index
),
consumed_counts AS (
    SELECT
        observation_index,
        COUNT(*) AS consumed_row_count
    FROM capstone.synthetic_sensor_messages_consumed_stage
    GROUP BY observation_index
)
SELECT
    s.observation_index,
    s.send_row_count,
    COALESCE(c.consumed_row_count, 0) AS consumed_row_count,
    s.send_row_count - COALESCE(c.consumed_row_count, 0) AS missing_row_count
FROM send_counts s
LEFT JOIN consumed_counts c
    ON s.observation_index = c.observation_index
WHERE s.send_row_count <> COALESCE(c.consumed_row_count, 0)
ORDER BY missing_row_count DESC, s.observation_index;

    SELECT sq.observation_index, sq.message_key
    FROM capstone.synthetic_sensor_messages_send_queue sq
    LEFT JOIN capstone.synthetic_sensor_messages_consumed_stage cs
        ON sq.message_key = cs.message_key
    WHERE cs.message_key IS NULL
)
SELECT
    observation_index,
    COUNT(*) AS missing_rows
FROM missing
GROUP BY observation_index
ORDER BY missing_rows DESC, observation_index;

SELECT
    pid,
    usename,
    application_name,
    state,
    wait_event_type,
    wait_event,
    now() - query_start AS runtime,
    query
FROM pg_stat_activity
WHERE datname = current_database()
  AND state <> 'idle'
ORDER BY query_start;


SELECT
    a.pid,
    a.state,
    a.wait_event_type,
    a.wait_event,
    now() - a.query_start AS runtime,
    pg_blocking_pids(a.pid) AS blocking_pids,
    a.query
FROM pg_stat_activity a
WHERE a.datname = current_database()
  AND a.state <> 'idle'
ORDER BY a.query_start;
